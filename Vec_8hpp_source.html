352  //-----------------------------------------------------------------------------
353  ALPAKA_NO_HOST_ACC_WARNING
354  template<
355  typename TFnObj,
356  std::size_t... TIndices>
357  ALPAKA_FN_HOST_ACC auto foldrByIndices(
358  TFnObj const & f,
359  std::integer_sequence<std::size_t, TIndices...> const & indices) const
360  {
361  alpaka::ignore_unused(indices);
362 
363  return
364  meta::foldr(
365  f,
366  ((*this)[TIndices])...);
367  }
368  //-----------------------------------------------------------------------------
369  ALPAKA_NO_HOST_ACC_WARNING
370  template<
371  typename TFnObj>
372  ALPAKA_FN_HOST_ACC auto foldrAll(
373  TFnObj const & f) const
374  {
375  return
376  foldrByIndices(
377  f,
378  IdxSequence());
379  }
380 // suppress strange warning produced by nvcc+MSVC in release mode
381 #if BOOST_COMP_MSVC
382  #pragma warning(push)

384 #endif

386  //! \return The product of all values.
387  ALPAKA_NO_HOST_ACC_WARNING
388  ALPAKA_FN_HOST_ACC auto prod() const
389  -> TVal
390  {
391  return foldrAll(std::multiplies<TVal>());
392  }
393 #if BOOST_COMP_MSVC
394  #pragma warning(pop)
395 #endif
396  //-----------------------------------------------------------------------------
397  //! \return The sum of all values.
398  ALPAKA_NO_HOST_ACC_WARNING
399  ALPAKA_FN_HOST_ACC auto sum() const
400  -> TVal
401  {
402  return foldrAll(std::plus<TVal>());
403  }

405  //! \return The min of all values.
406  ALPAKA_NO_HOST_ACC_WARNING
407  ALPAKA_FN_HOST_ACC auto min() const
408  -> TVal
409  {
410  return foldrAll(meta::min<TVal>());
411  }
412  //-----------------------------------------------------------------------------
413  //! \return The max of all values.
414  ALPAKA_NO_HOST_ACC_WARNING
415  ALPAKA_FN_HOST_ACC auto max() const

419  }

443  std::begin(m_data),
444  std::end(m_data))));
445  }
